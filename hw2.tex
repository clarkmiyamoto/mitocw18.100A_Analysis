\section*{Problem 1}
\textbf{Assumptions:} Let $F$ be an ordered field, and $x,y,z \in F$. Assume $x < 0$ and $y < z$.

\textbf{WTS:}
$xy > xz$
\\

\textbf{Proof:} If $x < 0$, we can rewrite $x = - \tilde x$ s.t. $\tilde x > 0$. So starting w/ the inequality between $y < z$:
\begin{align}
	y & < z\\
	\tilde x y & < \tilde x 	z\\
	- \tilde x y & > - \tilde x z\\
	x y & > x z
\end{align}
QED.
